    **2. Fix deactivateToken JS handler in session_detail.html** (lines 584-592):

    Replace the existing handler block (lines 584-592) with a proper fetch POST:
    ```javascript
    var deactivateBtn = document.querySelector('[data-action="deactivateToken"]');
    if (deactivateBtn) {
        deactivateBtn.addEventListener('click', function() {
            if (!confirm('Token wirklich deaktivieren?')) return;
            var tokenId = this.getAttribute('data-token-id');
            fetch('/finanzberatung/sessions/' + sessionId + '/deactivate-token', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'X-CSRFToken': document.querySelector('meta[name="csrf-token"]').getAttribute('content')
                },
                body: JSON.stringify({token_id: parseInt(tokenId)})
            })
            .then(function(response) { return response.json(); })
            .then(function(data) {
                if (data.success) {
                    window.location.reload();
                } else {
                    alert(data.message || 'Fehler beim Deaktivieren des Tokens.');
                }
            })
            .catch(function() {
                alert('Fehler beim Deaktivieren des Tokens.');
            });
        });
    }
    ```
